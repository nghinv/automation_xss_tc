COMM_login
echo ++ COMM_login
setSpeed 2000
echo INFO:: Logout first to make sure that no logged in session is alive in the browser
include COMM_logout.html
openAndWait /portal/login
windowMaximize
refreshAndWait
waitForElementPresent //input[contains(@id,'username')]
type //input[contains(@id,'username')] ${UserName}
type //input[@id='UIPortalLoginFormControl'] ${UserPassword}
clickAndWait css=button.button
openAndWait /portal/login?initialURI=/portal/intranet&username=${UserName}&password=${UserPassword}
echo -- COMM_login
