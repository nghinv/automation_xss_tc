COMM_FORUM_add_forum
echo ++ COMM_FORUM_add_forum
include COMM_get_timestamp.html
waitForElementPresent link=Forums
click link=Forums
waitForElementPresent link=Add Forum
click link=Add Forum
setSpeed 3000
waitForElementPresent //input[@id='ForumTitle']
type //input[@id='ForumTitle'] title${timestamp}
setSpeed 2000
type //input[@id='ForumOrder'] 1
type //textarea[@id='Description'] description${timestamp}
click link=Moderation Options
setSpeed 2000
storeEval window.jjQuery("#moderationOptions-tab").find('#AutoAddEmailNotify').attr('checked',false);
waitForElementPresent //textarea[@id='NotifyWhenAddPost']
type //textarea[@id='NotifyWhenAddPost'] post${timestamp}@exoplatform.com
type //textarea[@id='NotifyWhenAddTopic'] topic${timestamp}@exoplatform.com
echo -- COMM_FORUM_add_forum
